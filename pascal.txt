program poscal; //Название программы
var	W1,non,pol,j,i: integer;//целочисленные переменные столбоцов второго статичного массива,
                            //строки и столбцы со счетчиками для перебора двумерных массивово
const N = 4; //Константа массива
const M = 4; //Константа массива
var	a,b: array[0..N,0..M] of integer; //Двумерный статичный массив целочисленного типа
var f: boolean; //Логиская переменная 

begin // начало главной функции
    W1 := 0; //Инициализация счетчика для столбцов массива b
    f := true; //Инициализация логической переменной
    for i:=0 to N do begin      //
		for j:= 0 to M do       //
			begin               //
			    write(' ');     //
			    read(a[i,j]);   // Перебор массива a с присвоением значений для каждого элемента 
			end;                //
		writeln();              //
		end;                    //
	writeln();                  //


	for non:=0 to N do begin                                //
	    W1 := 0;                                            // Перебор массива 
		for pol:= 0 to M do	begin                           //  
			f:= true;                                       //
				for i:= 0 to W1 do begin                    // с проверкой для каждого значения
						if a[non,pol] = b[non,i] then begin // на наличие повторений
                            f:= false;                      //
                        end;                                // 
					end;                                    //
				if f = true then begin                      // с добавлением непвторяющегося элемента 
				    b[non,W1] := a[non,pol];                //
					W1 := W1 + 1;                           // в коцне строки массива b цикл перебирает все значения
				end;                                        // строки массива a и проверяет на уникальность элемента  
			end;	                                        //
		end;                                                //
		
	for i:=0 to N do begin      //
		for j:= 0 to M do begin //
			write(b[i,j]:3);    // вывод итогового массива b с устраненным повторением не нулевых элементов
			end;                //
		writeln;                //
		end;                    //
end.
